  <script>
    // Hardcoded student data
    const students = [
      {roll: 'F500490025', mother: 'SUREKHA', pdf: 'result_101.pdf'},
      {roll: '102', mother: 'Anita', pdf: 'result_102.pdf'},
      {roll: '103', mother: 'Meena', pdf: 'result_103.pdf'}
    ];

    function checkResult() {
      const roll = document.getElementById('roll').value.trim();
      const mother = document.getElementById('mother').value.trim();
      
      const student = students.find(s => s.roll === roll && s.mother.toLowerCase() === mother.toLowerCase());
      
      if(student){
        window.open(student.pdf, '_blank');
      } else {
        alert('No result found! Check your details.');
      }
    }
